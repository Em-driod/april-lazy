// More robust version that handles edge cases const counterElement = document.getElementById('counterValue'); const decreaseBtn = document.getElementById('decreaseBtn'); let counter = parseInt(counterElement.textContent); // Ensure counter is a valid number if (isNaN(counter)) { counter = 0; counterElement.textContent = counter; } function decreaseCounter() { if (counter > 0) { counter--; counterElement.textContent = counter; } else { // Optional: Show an alert or disable the button console.log('Counter cannot go below 0'); decreaseBtn.disabled = true; } } decreaseBtn.addEventListener('click', decreaseCounter);
